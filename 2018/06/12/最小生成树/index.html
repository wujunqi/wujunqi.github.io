// иҫ№
template<typename Weight>
class Edge{
private:
    int a,b;    // иҫ№зҡ„дёӨдёӘз«ҜзӮ№
    Weight weight;  // иҫ№зҡ„жқғеҖј

public:
    // жһ„йҖ еҮҪж•°
    Edge(int a, int b, Weight weight){
        this->a = a;
        this->b = b;
        this->weight = weight;
    }
    // з©әзҡ„жһ„йҖ еҮҪж•°, жүҖжңүзҡ„жҲҗе‘ҳеҸҳйҮҸйғҪеҸ–й»ҳи®ӨеҖј
    Edge(){}

    ~Edge(){}

    int v(){ return a;} // иҝ”еӣһз¬¬дёҖдёӘйЎ¶зӮ№
    int w(){ return b;} // иҝ”еӣһз¬¬дәҢдёӘйЎ¶зӮ№
    Weight wt(){ return weight;}    // иҝ”еӣһжқғеҖј

    // з»ҷе®ҡдёҖдёӘйЎ¶зӮ№, иҝ”еӣһеҸҰдёҖдёӘйЎ¶зӮ№
    int other(int x){
        assert( x == a || x == b );
        return x == a ? b : a;
    }

    // иҫ“еҮәиҫ№зҡ„дҝЎжҒҜ
    friend ostream& operator<<(ostream &os, const Edge &e){
        os<<e.a<<"-"<<e.b<<": "<<e.weight;
        return os;
    }

    // иҫ№зҡ„еӨ§е°ҸжҜ”иҫғ, жҳҜеҜ№иҫ№зҡ„жқғеҖјзҡ„еӨ§е°ҸжҜ”иҫғ
    bool operator<(Edge<Weight>& e){
        return weight < e.wt();
    }
    bool operator<=(Edge<Weight>& e){
        return weight <= e.wt();
    }
    bool operator>(Edge<Weight>& e){
        return weight > e.wt();
    }
    bool operator>=(Edge<Weight>& e){
        return weight >= e.wt();
    }
    bool operator==(Edge<Weight>& e){
        return weight == e.wt();
    }
};

// зЁ еҜҶеӣҫ - йӮ»жҺҘзҹ©йҳө
template <typename Weight>
class DenseGraph{

private:
    int n, m;       // иҠӮзӮ№ж•°е’Ңиҫ№ж•°
    bool directed;  // жҳҜеҗҰдёәжңүеҗ‘еӣҫ
    vector<vector<Edge<Weight> *>> g;   // еӣҫзҡ„е…·дҪ“ж•°жҚ®

public:
    // жһ„йҖ еҮҪж•°
    DenseGraph( int n , bool directed){
        assert( n >= 0 );
        this->n = n;
        this->m = 0;
        this->directed = directed;
        // gеҲқе§ӢеҢ–дёәn*nзҡ„зҹ©йҳө, жҜҸдёҖдёӘg[i][j]жҢҮеҗ‘дёҖдёӘиҫ№зҡ„дҝЎжҒҜ, еҲқе§ӢеҢ–дёәNULL
        g = vector<vector<Edge<Weight> *>>(n, vector<Edge<Weight> *>(n, NULL));
    }

    // жһҗжһ„еҮҪж•°
    ~DenseGraph(){

        for( int i = 0 ; i < n ; i ++ )
            for( int j = 0 ; j < n ; j ++ )
                if( g[i][j] != NULL )
                    delete g[i][j];
    }

    int V(){ return n;} // иҝ”еӣһиҠӮзӮ№дёӘж•°
    int E(){ return m;} // иҝ”еӣһиҫ№зҡ„дёӘж•°

    // еҗ‘еӣҫдёӯж·»еҠ дёҖдёӘиҫ№, жқғеҖјдёәweight
    void addEdge( int v, int w , Weight weight ){
        assert( v >= 0 && v < n );
        assert( w >= 0 && w < n );

        // еҰӮжһңд»ҺvеҲ°wе·Із»Ҹжңүиҫ№, еҲ йҷӨиҝҷжқЎиҫ№
        if( hasEdge( v , w  ) ){
            delete  g[v][w];
            if( v != w && !directed )
                delete g[w][v];
            m --;
        }

        g[v][w] = new Edge<Weight>(v, w, weight);
        if( v != w && !directed )
            g[w][v] = new Edge<Weight>(w, v, weight);
        m ++;
    }

    // йӘҢиҜҒеӣҫдёӯжҳҜеҗҰжңүд»ҺvеҲ°wзҡ„иҫ№
    bool hasEdge( int v , int w ){
        assert( v >= 0 && v < n );
        assert( w >= 0 && w < n );
        return g[v][w] != NULL;
    }

    // жҳҫзӨәеӣҫзҡ„дҝЎжҒҜ
    void show(){

        for( int i = 0 ; i < n ; i ++ ){
            for( int j = 0 ; j < n ; j ++ )
                if( g[i][j] )
                    cout<<g[i][j]->wt()<<"\t";
                else
                    cout<<"NULL\t";
            cout<<endl;
        }
    }

    // йӮ»иҫ№иҝӯд»ЈеҷЁ, дј е…ҘдёҖдёӘеӣҫе’ҢдёҖдёӘйЎ¶зӮ№,
    // иҝӯд»ЈеңЁиҝҷдёӘеӣҫдёӯе’ҢиҝҷдёӘйЎ¶зӮ№еҗ‘иҝһзҡ„жүҖжңүиҫ№
    class adjIterator{
    private:
        DenseGraph &G;  // еӣҫGзҡ„еј•з”Ё
        int v;
        int index;

    public:
        // жһ„йҖ еҮҪж•°
        adjIterator(DenseGraph &graph, int v): G(graph){
            this->v = v;
            this->index = -1;   // зҙўеј•д»Һ-1ејҖе§Ӣ, еӣ дёәжҜҸж¬ЎйҒҚеҺҶйғҪйңҖиҰҒи°ғз”ЁдёҖж¬Ўnext()
        }

        ~adjIterator(){}

        // иҝ”еӣһеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„з¬¬дёҖдёӘиҫ№
        Edge<Weight>* begin(){
            // зҙўеј•д»Һ-1ејҖе§Ӣ, еӣ дёәжҜҸж¬ЎйҒҚеҺҶйғҪйңҖиҰҒи°ғз”ЁдёҖж¬Ўnext()
            index = -1;
            return next();
        }

        // иҝ”еӣһеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„дёӢдёҖдёӘиҫ№
        Edge<Weight>* next(){
            // д»ҺеҪ“еүҚindexејҖе§Ӣеҗ‘еҗҺжҗңзҙў, зӣҙеҲ°жүҫеҲ°дёҖдёӘg[v][index]дёәtrue
            for( index += 1 ; index < G.V() ; index ++ )
                if( G.g[v][index] )
                    return G.g[v][index];
            // иӢҘжІЎжңүйЎ¶зӮ№е’ҢvзӣёиҝһжҺҘ, еҲҷиҝ”еӣһNULL
            return NULL;
        }

        // жҹҘзңӢжҳҜеҗҰе·Із»Ҹиҝӯд»Је®ҢдәҶеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„жүҖжңүиҫ№
        bool end(){
            return index >= G.V();
        }
    };
};

// зЁҖз–Ҹеӣҫ - йӮ»жҺҘиЎЁ
template<typename Weight>
class SparseGraph{

private:
    int n, m;       // иҠӮзӮ№ж•°е’Ңиҫ№ж•°
    bool directed;  // жҳҜеҗҰдёәжңүеҗ‘еӣҫ
    vector<vector<Edge<Weight> *> > g;   // еӣҫзҡ„е…·дҪ“ж•°жҚ®

public:
    // жһ„йҖ еҮҪж•°
    SparseGraph( int n , bool directed){
        assert(n >= 0);
        this->n = n;
        this->m = 0;    // еҲқе§ӢеҢ–жІЎжңүд»»дҪ•иҫ№
        this->directed = directed;
        // gеҲқе§ӢеҢ–дёәnдёӘз©әзҡ„vector, иЎЁзӨәжҜҸдёҖдёӘg[i]йғҪдёәз©ә, еҚіжІЎжңүд»»е’Ңиҫ№
        g = vector<vector<Edge<Weight> *> >(n, vector<Edge<Weight> *>());
    }

    // жһҗжһ„еҮҪж•°
    ~SparseGraph(){
        for( int i = 0 ; i < n ; i ++ )
            for( int j = 0 ; j < g[i].size() ; j ++ )
                delete g[i][j];
    }

    int V(){ return n;} // иҝ”еӣһиҠӮзӮ№дёӘж•°
    int E(){ return m;} // иҝ”еӣһиҫ№зҡ„дёӘж•°

    // еҗ‘еӣҫдёӯж·»еҠ дёҖдёӘиҫ№, жқғеҖјдёәweight
    void addEdge( int v, int w , Weight weight){
        assert( v >= 0 && v < n );
        assert( w >= 0 && w < n );

        // жіЁж„Ҹ, з”ұдәҺеңЁйӮ»жҺҘиЎЁзҡ„жғ…еҶө, жҹҘжүҫжҳҜеҗҰжңүйҮҚиҫ№йңҖиҰҒйҒҚеҺҶж•ҙдёӘй“ҫиЎЁ
        // жҲ‘д»¬зҡ„зЁӢеәҸе…Ғи®ёйҮҚиҫ№зҡ„еҮәзҺ°

        g[v].push_back(new Edge<Weight>(v, w, weight));
        if( v != w && !directed )
            g[w].push_back(new Edge<Weight>(w, v, weight));
        m ++;
    }

    // йӘҢиҜҒеӣҫдёӯжҳҜеҗҰжңүд»ҺvеҲ°wзҡ„иҫ№
    bool hasEdge( int v , int w ){
        assert( v >= 0 && v < n );
        assert( w >= 0 && w < n );
        for( int i = 0 ; i < g[v].size() ; i ++ )
            if( g[v][i]->other(v) == w )
                return true;
        return false;
    }

    // жҳҫзӨәеӣҫзҡ„дҝЎжҒҜ
    void show(){

        for( int i = 0 ; i < n ; i ++ ){
            cout<<"vertex "<<i<<":\t";
            for( int j = 0 ; j < g[i].size() ; j ++ )
                cout<<"( to:"<<g[i][j]->w()<<",wt:"<<g[i][j]->wt()<<")\t";
            cout<<endl;
        }
    }

    // йӮ»иҫ№иҝӯд»ЈеҷЁ, дј е…ҘдёҖдёӘеӣҫе’ҢдёҖдёӘйЎ¶зӮ№,
    // иҝӯд»ЈеңЁиҝҷдёӘеӣҫдёӯе’ҢиҝҷдёӘйЎ¶зӮ№еҗ‘иҝһзҡ„жүҖжңүиҫ№
    class adjIterator{
    private:
        SparseGraph &G; // еӣҫGзҡ„еј•з”Ё
        int v;
        int index;

    public:
        // жһ„йҖ еҮҪж•°
        adjIterator(SparseGraph &graph, int v): G(graph){
            this->v = v;
            this->index = 0;
        }

        ~adjIterator(){}

        // иҝ”еӣһеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„з¬¬дёҖдёӘиҫ№
        Edge<Weight>* begin(){
            index = 0;
            if( G.g[v].size() )
                return G.g[v][index];
            // иӢҘжІЎжңүйЎ¶зӮ№е’ҢvзӣёиҝһжҺҘ, еҲҷиҝ”еӣһNULL
            return NULL;
        }

        // иҝ”еӣһеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„дёӢдёҖдёӘиҫ№
        Edge<Weight>* next(){
            index += 1;
            if( index < G.g[v].size() )
                return G.g[v][index];
            return NULL;
        }

        // жҹҘзңӢжҳҜеҗҰе·Із»Ҹиҝӯд»Је®ҢдәҶеӣҫGдёӯдёҺйЎ¶зӮ№vзӣёиҝһжҺҘзҡ„жүҖжңүйЎ¶зӮ№
        bool end(){
            return index >= G.g[v].size();
        }
    };
};

иҝҗиЎҢз»“жһңпјҡ

жңҖе°Ҹз”ҹжҲҗж ‘

жңҖе°Ҹз”ҹжҲҗж ‘жҳҜдёҖеүҜиҝһйҖҡеҠ жқғж— еҗ‘еӣҫдёӯдёҖжЈөжқғеҖјжңҖе°Ҹзҡ„з”ҹжҲҗж ‘гҖӮ
еҗ„дёӘиҠӮзӮ№д№Ӣй—ҙиҝһйҖҡпјҢиҝһйҖҡжҖ»иҙ№з”ЁжңҖе°ҸгҖӮ

зӣёе…іеә”з”Ё

  • з”өзјҶеёғзәҝи®ҫи®Ў
  • зҪ‘з»ңи®ҫи®Ў
  • з”өи·Ҝи®ҫи®Ў

жңҖе°Ҹз”ҹжҲҗж ‘дё»иҰҒй’ҲеҜ№пјҡ

  • й’ҲеҜ№еёҰжқғж— еҗ‘еӣҫ
  • й’ҲеҜ№иҝһйҖҡеӣҫ
  • еҜ№дәҺдёҚиҝһйҖҡзҡ„еӣҫпјҢеҸҜд»ҘжұӮжүҖжңүжҳҜиҝһйҖҡеҲҶйҮҸзҡ„жңҖе°Ҹз”ҹжҲҗж ‘еҪўжҲҗжңҖе°ҸжЈ®жһ—гҖӮ

жӯҘйӘӨ

  • жүҫ V-1 жқЎиҫ№
  • иҝһжҺҘVдёӘйЎ¶зӮ№
  • жҖ»жқғеҖјжңҖе°Ҹ

еҲҮеҲҶе®ҡзҗҶпјҡCut Property

еҲҮеҲҶзҡ„е®ҡд№ү

жҠҠеӣҫдёӯзҡ„иҠӮзӮ№еҲҶжҲҗдёӨйғЁеҲҶпјҢжҲҗдёәдёҖдёӘеҲҮеҲҶпјҲcutпјү

и“қиүІе’ҢзәўиүІзҡ„йғЁеҲҶеҪўжҲҗдәҶдёҖдёӘеҲҮеҲҶгҖӮ

жЁӘеҲҮиҫ№зҡ„е®ҡд№ү

еҰӮжһңдёҖдёӘиҫ№зҡ„дёӨдёӘз«ҜзӮ№пјҢеұһдәҺеҲҮеҲҶ(Cut)дёҚеҗҢзҡ„дёӨиҫ№пјҢиҝҷдёӘиҫ№з§°дёәжЁӘеҲҮиҫ№(Crossing Edge)гҖӮ

еҲҮеҲҶе®ҡзҗҶпјҡ

з»ҷе®ҡд»»ж„ҸеҲҮеҲҶпјҢжЁӘеҲҮиҫ№дёӯжқғеҖјжңҖе°Ҹзҡ„иҫ№еҝ…з„¶еұһдәҺжңҖе°Ҹз”ҹжҲҗж ‘гҖӮ

Lazy Prim

  • жүҖжңүиҫ№дёӯйҖүеҸ–еҮәv-1жқЎгҖӮ
  • жүҫеҮәеӣӣжқЎжЁӘеҲҮиҫ№дёӯжңҖе°Ҹзҡ„
    • жңҖе°Ҹе ҶиҝӣиЎҢе®һзҺ°
  • жҠҠжңҖе°Ҹиҫ№еҠ е…ҘеҗҺгҖӮиҝӣиЎҢж–°зҡ„еҲҮеҲҶ
  • дёҚж–ӯеҠ е…ҘжңҖе°ҸжЁӘеҲҮиҫ№зҡ„зӮ№пјҢзӣҙеҲ°жүҖжңүйЎ¶зӮ№йғҪиҝӣе…ҘпјҲзӣҙеҲ°жңҖеҗҺжүҖжңүиҠӮзӮ№иў«и®ҝй—®иҝҮпјүгҖӮ
  • жҮ’жғ°пјҡиҷҪз„¶дёҚжҳҜжЁӘеҲҮиҫ№пјҢдҪҶжҳҜжІЎжңүиў«жү”жҺүгҖӮ

// дҪҝз”ЁPrimз®—жі•жұӮеӣҫзҡ„жңҖе°Ҹз”ҹжҲҗж ‘
template<typename Graph, typename Weight>
class LazyPrimMST{

private:
    Graph &G;                   // еӣҫзҡ„еј•з”Ё
    MinHeap<Edge<Weight>> pq;   // жңҖе°Ҹе Ҷ, з®—жі•иҫ…еҠ©ж•°жҚ®з»“жһ„
    bool *marked;               // ж Үи®°ж•°з»„, еңЁз®—жі•иҝҗиЎҢиҝҮзЁӢдёӯж Үи®°иҠӮзӮ№iжҳҜеҗҰиў«и®ҝй—®
    vector<Edge<Weight>> mst;   // жңҖе°Ҹз”ҹжҲҗж ‘жүҖеҢ…еҗ«зҡ„жүҖжңүиҫ№
    Weight mstWeight;           // жңҖе°Ҹз”ҹжҲҗж ‘зҡ„жқғеҖј

    // и®ҝй—®иҠӮзӮ№v
    void visit(int v){

        assert( !marked[v] );
        marked[v] = true;

        // е°Ҷе’ҢиҠӮзӮ№vзӣёиҝһжҺҘзҡ„жүҖжңүжңӘи®ҝй—®зҡ„иҫ№ж”ҫе…ҘжңҖе°Ҹе Ҷдёӯ
        typename Graph::adjIterator adj(G,v);
        for( Edge<Weight>* e = adj.begin() ; !adj.end() ; e = adj.next() )
            if( !marked[e->other(v)] )
                pq.insert(*e);
    }

public:
    // жһ„йҖ еҮҪж•°, дҪҝз”ЁPrimз®—жі•жұӮеӣҫзҡ„жңҖе°Ҹз”ҹжҲҗж ‘
    LazyPrimMST(Graph &graph):G(graph), pq(MinHeap<Edge<Weight>>(graph.E())){

        // з®—жі•еҲқе§ӢеҢ–
        marked = new bool[G.V()];
        for( int i = 0 ; i < G.V() ; i ++ )
            marked[i] = false;
        mst.clear();

        // Lazy Prim
        visit(0);
        while( !pq.isEmpty() ){
            // дҪҝз”ЁжңҖе°Ҹе ҶжүҫеҮәе·Із»Ҹи®ҝй—®зҡ„иҫ№дёӯжқғеҖјжңҖе°Ҹзҡ„иҫ№
            Edge<Weight> e = pq.extractMin();
            // еҰӮжһңиҝҷжқЎиҫ№зҡ„дёӨз«ҜйғҪе·Із»Ҹи®ҝй—®иҝҮдәҶ, еҲҷжү”жҺүиҝҷжқЎиҫ№
            if( marked[e.v()] == marked[e.w()] )
                continue;
            // еҗҰеҲҷ, иҝҷжқЎиҫ№еҲҷеә”иҜҘеӯҳеңЁеңЁжңҖе°Ҹз”ҹжҲҗж ‘дёӯ
            mst.push_back( e );

            // и®ҝй—®е’ҢиҝҷжқЎиҫ№иҝһжҺҘзҡ„иҝҳжІЎжңүиў«и®ҝй—®иҝҮзҡ„иҠӮзӮ№
            if( !marked[e.v()] )
                visit( e.v() );
            else
                visit( e.w() );
        }

        // и®Ўз®—жңҖе°Ҹз”ҹжҲҗж ‘зҡ„жқғеҖј
        mstWeight = mst[0].wt();
        for( int i = 1 ; i < mst.size() ; i ++ )
            mstWeight += mst[i].wt();
    }

    // жһҗжһ„еҮҪж•°
    ~LazyPrimMST(){
        delete[] marked;
    }

    // иҝ”еӣһжңҖе°Ҹз”ҹжҲҗж ‘зҡ„жүҖжңүиҫ№
    vector<Edge<Weight>> mstEdges(){
        return mst;
    };

    // иҝ”еӣһжңҖе°Ҹз”ҹжҲҗж ‘зҡ„жқғеҖј
    Weight result(){
        return mstWeight;
    };
};

// Kruskalз®—жі•
template <typename Graph, typename Weight>
class KruskalMST{

private:
    vector<Edge<Weight>> mst;   // жңҖе°Ҹз”ҹжҲҗж ‘жүҖеҢ…еҗ«зҡ„жүҖжңүиҫ№
    Weight mstWeight;           // жңҖе°Ҹз”ҹжҲҗж ‘зҡ„жқғеҖј

public:
    // жһ„йҖ еҮҪж•°, дҪҝз”ЁKruskalз®—жі•и®Ўз®—graphзҡ„жңҖе°Ҹз”ҹжҲҗж ‘
    KruskalMST(Graph &graph){

        // е°Ҷеӣҫдёӯзҡ„жүҖжңүиҫ№еӯҳж”ҫеҲ°дёҖдёӘжңҖе°Ҹе Ҷдёӯ
        MinHeap<Edge<Weight>> pq( graph.E() );
        for( int i = 0 ; i < graph.V() ; i ++ ){
            typename Graph::adjIterator adj(graph,i);
            for( Edge<Weight> *e = adj.begin() ; !adj.end() ; e = adj.next() )
                //еҸӘеӯҳдёҖжқЎиҫ№дёӯзҡ„дёҖдёӘиҠӮзӮ№
                if( e->v() < e->w() )
                    pq.insert(*e);
        }

        // еҲӣе»әдёҖдёӘе№¶жҹҘйӣҶ, жқҘжҹҘзңӢе·Із»Ҹи®ҝй—®зҡ„иҠӮзӮ№зҡ„иҒ”йҖҡжғ…еҶө
        UnionFind uf = UnionFind(graph.V());
        while( !pq.isEmpty() && mst.size() < graph.V() - 1 ){

            // д»ҺжңҖе°Ҹе Ҷдёӯдҫқж¬Ўд»Һе°ҸеҲ°еӨ§еҸ–еҮәжүҖжңүзҡ„иҫ№
            Edge<Weight> e = pq.extractMin();
            // еҰӮжһңиҜҘиҫ№зҡ„дёӨдёӘз«ҜзӮ№жҳҜиҒ”йҖҡзҡ„, иҜҙжҳҺеҠ е…ҘиҝҷжқЎиҫ№е°Ҷдә§з”ҹзҺҜ, жү”жҺүиҝҷжқЎиҫ№
            if( uf.isConnected( e.v() , e.w() ) )
                continue;

            // еҗҰеҲҷ, е°ҶиҝҷжқЎиҫ№ж·»еҠ иҝӣжңҖе°Ҹз”ҹжҲҗж ‘, еҗҢж—¶ж Үи®°иҫ№зҡ„дёӨдёӘз«ҜзӮ№иҒ”йҖҡ
            mst.push_back( e );
            uf.unionElements( e.v() , e.w() );
        }

        mstWeight = mst[0].wt();
        for( int i = 1 ; i < mst.size() ; i ++ )
            mstWeight += mst[i].wt();
    }

    ~KruskalMST(){ }

    // иҝ”еӣһжңҖе°Ҹз”ҹжҲҗж ‘зҡ„жүҖжңүиҫ№
    vector<Edge<Weight>> mstEdges(){
        return mst;
    };

    // иҝ”еӣһжңҖе°Ҹз”ҹжҲҗж ‘зҡ„жқғеҖј
    Weight result(){
        return mstWeight;
    };
};
